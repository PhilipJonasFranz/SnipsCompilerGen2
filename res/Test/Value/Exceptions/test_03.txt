struct Exception {
	int code;
}

int checkThreshold(int* arr, int t) signals Exception {
	int max = 0;
	int s = hsize(arr);
	for (int i = 0; i < s; i++) {
		int v = *(arr + i);
		if (v >= t) {
			signal Exception::(v);
		}
		else {
			if (v > max) max = v;
		}
	}
	
	return max;
}

int main(int t) {
	int* arr = (int*) resv(5);
	*arr = {6, 1, 7, 5, 9};
	
	int exit = 0;
	try {
		exit = checkThreshold(arr, t);
	} watch (Exception e) {
		exit = e.code;
	}
	
	return exit;
}
