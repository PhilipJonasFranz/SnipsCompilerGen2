}

struct Exception {
	int i0;
	int s0;
}

int get(int* arr, int i, int s) signals Exception {
	if (i > 2) {
		signal Exception::(i, s);
	}
	else {
		return *(arr + i);
	}
}

int throw() signals ContextException {
	signal ContextException::(99);
}

int main(int i) {
	int [3] arr = {5, 3, 8};
	
	int r = 0;
	try {
		throw();
		r = get(&arr, i, 3);
	} watch (Exception e) {
		r = (e.i0 << 1) + e.s0;
	} watch (ContextException e) {
		r = e.code;
	}
	
	return r;
}
