int fib(int n) {
	int* arr = resv(n);
	
	int c = 0;
	for [int i : arr, n] {
		if (c < 2) i = 1;
		else i = *(arr + c - 1) + *(arr + c - 2);
		c++;
	}
	
	return *(arr + n - 1);
}

int main(int n) {
	return fib(n);
