DESCRIPTION
This Test aims to verify the correct functionality of inline calls that have to load parameters
from the stack and registers.
SOURCE
int foo(int a, int e) {
	return a * e;
}

int main(int a, int b, int c, int d, int e) {
	return foo(a, e) + b + (d * c);
}
TESTCASES
1 2 3 4 5 19
1 0 1 0 1 1
OUTPUT
.version 0

.data
NULL: .word 0
HEAP_START: .word 0

.text
    b main

.global foo
foo:                                         /* Function: foo */
    mul r0, r0, r1
    bx lr

.global main
main:                                        /* Function: main */
    push { fp, lr }
    mov fp, sp
    mov r4, r1
    mov r5, r2
    ldr r1, [fp, #8]                         /* Evaluate Expression */
    bl foo                                   /* Call foo */
    add r0, r0, r4
    ldr r1, [fp, #12]
    mul r2, r1, r5
    add r0, r0, r2
    mov sp, fp
    ldmfd sp!, {fp, pc}
