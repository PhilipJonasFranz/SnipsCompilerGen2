DESCRIPTION
This Test aims to verify the correct behaviour of the main function when parameters are
passed through the stack and registers, as well as the access through them in following
statements.
SOURCE
int main(int a, int b, int c, int d, int e) {
	return (a * e) + b + (d * c);
}
TESTCASES
1 2 3 4 5 19
1 0 1 0 1 1
OUTPUT
.version 0

.data
NULL: .word 0
HEAP_START: .word 0

.text
    b main

.global main
main:                                        /* Function: main */
    mov r4, r1
    mov r5, r2
    mov r1, r0                               /* Evaluate Expression */
    ldr r2, [sp]
    mla r0, r1, r2, r4
    ldr r1, [sp, #4]
    mul r2, r1, r5
    add r0, r0, r2
    bx lr
