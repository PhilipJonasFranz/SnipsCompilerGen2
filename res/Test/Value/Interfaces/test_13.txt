interface Getter<K> {

	K get<K>();

}

interface GetSet<V> : Getter<V> {

	void set<V>(V val);

}

struct A<K> : GetSet<K> {
	K val;
	
	K get<K>() {
		return self->val;
	}
	
	K set<K>(K val) {
		self->val = val;
	}
}

int main() {
	GetSet<int> gs = init<>(A<int>::(10));
	
	int s = gs.get();
	gs.set(20);
	
	return s + gs.get();
}
