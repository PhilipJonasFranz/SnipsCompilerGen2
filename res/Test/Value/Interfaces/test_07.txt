DESCRIPTION
Test the correct functionality and behaviour when an interface is passed as parameter.
SOURCE
interface Inter {

	int get();

}

struct X : Inter {

	int val;
	
	int get() {
		return self->val;
	}

}

struct Y : Inter {
	int val;
	
	int get() {
		return self->val * 2;
	}

}

int process(Inter i) {
	return i.get();
}

int main(bool b) {
	X x = X::(25);
	Inter i = &x;
	
	Y y = Y::(5);
	Inter i0 = &y;
	
	return process((b)? i0 : i);
}
