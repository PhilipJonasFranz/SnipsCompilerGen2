#include<linked_list.sn>

struct X<T> {
	
	T val;
	
	T get() {
		return self->val;
	}
	
	void set(T nVal) {
		self->val = nVal;
	}
	
	static X<T> create<T>(T val) {
		return X<T>::(val);
	}
}

int main() {
	X<int>* x = init<>(X::create<int>(10));
	LinkedList<X<int>*>* list = init<>(LinkedList::create<X<int>*>(null));
	
	list->add(x);
	
	x->set(12);
	
	return list->get(0)->get();
}
