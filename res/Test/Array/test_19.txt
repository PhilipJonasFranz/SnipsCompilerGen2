int __vec_cross(int [3] v, int [3] w) {
	int sum = 0;
	for (int i = 0; i < 3; i = i + 1) {
		sum = sum + (v [i] * w [i]);
	}
	return sum;
}

int main() {
	int [3] [3] m = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
	int [3] [3] n = {{1, 4, 7}, {2, 5, 8}, {3, 6, 9}};

	return __vec_cross(m [0], n [0]);
}
