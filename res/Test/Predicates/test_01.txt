void sort(int* arr, int size, func (int, int) -> bool pred) {
	if (size > 1) {
		while (true) {
			bool sorted = true;
			for (int i = 1; i < size; i++) {
				int a0 = *(arr + i - 1);
				int a1 = *(arr + i);
				if (pred(a1, a0)) {
					sorted = false;
					*(arr + i - 1) = a1;
					*(arr + i) = a0;
				}
			}
		
			if (sorted) break;
		}
	}
	return;
}
