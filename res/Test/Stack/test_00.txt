DESCRIPTION
This Test aims to verify the correct behaviour of the main function when parameters
are passed throught the stack and registers, as well as a simple use case of the parameters.
SOURCE
int main(int a, int b, int c, int d) {
	return a + b + c - d;
}
TESTCASES
1 2 3 4 2
2 2 2 2 4
3 6 2 5 6
OUTPUT
/* --test_00.txt */
.data
HEAP_START: .word 0
    
.text
main:                         /* Function: main */
    push { r3, r4, r5, fp, lr }
    mov fp, sp
    mov r3, r0
    mov r4, r1
    mov r5, r2
    add r1, r3, r4
    add r1, r1, r5
    ldr r2, [fp, #20]
    sub r0, r1, r2
    mov sp, fp
    pop { r3, r4, r5, fp, lr }
    bx lr
    
.L0_HEAP_START: .word HEAP_START
