#include<linked_list.sn>

int main(int x, int y) {
	int[2] v0 = {10, 20};
	int[2] v1 = {5, 8};
	int[2] v2 = {3, 9};

	List<int*>* l = create<int*>();
	add<int*>(l, &v0);
	add<int*>(l, &v1);
	add<int*>(l, &v2);
	
	if (contains<int*>(l, &v0) && contains<int*>(l, &v1) && contains<int*>(l, &v2)) {
		int* r = get<int*>(l, x);
		destroy<int*>(l);
		return *(r + y);
	}
	else return 0;
}
