struct Capsule<T> {
	T cap;
}

struct Data<T> {
	T value;
	Capsule<T> c;
}

Data<T>* create<T>() {
	Data<T>* pointer = (Data<T>*) resv(sizeof(Data<T>));
	return pointer;
}

int main(int x) {
	Data<int[2]>* data = create<int[2]>();
	*data = Data<int[2]>::({x, x * 2}, Capsule<int[2]>::({3, 4}));
	return data->value [1] + data->c.cap [1];
}
