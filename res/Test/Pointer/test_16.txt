int get(int*[2] p, int off) {
	return *(p [0] + off) - *(p [1] + off);
}

int main(int off) {
	int [2] [2] v = {{5, 3}, {9, 1}};
	int [2] [2] w = {{1, 2}, {4, 3}};
	int*[2] p = {&v, &w};
	return get(p, off);
}
